using System;
using System.Text;

class Program
{
    static void Main()
    {
        string input = "hello world from c sharp";
        StringBuilder result = new StringBuilder();

        for (int i = 0; i < input.Length; i++)
        {
            if (input[i] == ' ')
            {
                result.Append("%20");
            }
            else
            {
                result.Append(input[i]);
            }
        }

        Console.WriteLine(result.ToString());
    }
}
